CREATE TABLE GUEST 
(
  ID NUMBER NOT NULL 
, NAME VARCHAR2(40) NOT NULL 
, BIRTH_DATE DATE NOT NULL 
, BIRTH_CITY VARCHAR2(40) NOT NULL 
, TAX_NUMBER NUMBER NOT NULL 
, SSN VARCHAR2(10) NOT NULL 
, ADDRESS VARCHAR2(50) NOT NULL 
, CONSTRAINT GUEST_PK PRIMARY KEY 
  (
    ID 
  )
  ENABLE 
);

CREATE SEQUENCE GUEST_SEQ;

CREATE TRIGGER GUEST_TRG 
BEFORE INSERT ON GUEST 
FOR EACH ROW 
BEGIN
  <<COLUMN_SEQUENCES>>
  BEGIN
    IF INSERTING AND :NEW.ID IS NULL THEN
      SELECT GUEST_SEQ.NEXTVAL INTO :NEW.ID FROM SYS.DUAL;
    END IF;
  END COLUMN_SEQUENCES;
END;
